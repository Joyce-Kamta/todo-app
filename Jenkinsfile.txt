pipeline {

    agent any

    stages {

        stage('Récupération du code') {
            steps {
                echo "Build numéro : ${env.BUILD_NUMBER}"
                echo "Nom du job : ${env.JOB_NAME}"
                git branch: 'main',
                    url: 'https://github.com/Joyce-Kamta/todo-app.git'
            }
        }

        stage('Installation des dépendances') {
            steps {
                bat 'npm install'
            }
        }

        stage('Tests') {
            steps {
                bat 'npm test'
            }
        }

        stage('Résultats') {
            steps {
                echo 'Tests terminés avec succès !'
                echo "Build ${env.BUILD_NUMBER} réussi !"
            }
        }

    }

    post {
        success {
            echo '✅ Pipeline réussie — tout fonctionne !'
        }
        failure {
            echo '❌ Pipeline échouée — regarde les logs !'
        }
    }

}
